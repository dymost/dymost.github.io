Quartz 2D学习（一）基础绘制

  • 动画、Quartz 2D
动画

画一个基本图形步骤

  1. 获取图形上下文

    CGContextRef ctr = UIGraphicsGetCurrentContext();

  2. 划线，利用线组成图形

    // 创建线
    CGMutablePathRef path = CGPathCreateMutable();

    CGPathMoveToPoint(path, NULL, 50, 50);
    CGPathAddLineToPoint(path, NULL, 100, 50);
    CGPathAddLineToPoint(path, NULL, 100, 100);
    CGPathAddLineToPoint(path, NULL, 100, 100);

  3. 设置线的样式，图形的样式

    // 线宽
    CGContextSetLineWidth(ctr, 20);

    // 设置连接样式
    CGContextSetLineJoin(ctr, kCGLineJoinRound);

    // 设置顶角样式
    CGContextSetLineCap(ctr, kCGLineCapRound);

    // 设置填充颜色
    CGContextSetFillColorWithColor(ctr, [[UIColor blueColor] CGColor]);
    // 设置线的颜色
    CGContextSetStrokeColorWithColor(ctr, [[UIColor redColor] CGColor]);// 设置线的颜色

  1. 将路径添加到上下文

    CGContextAddPath(ctr, path);
    CGContext方式默认添加了路径

  2. 渲染上下文，将画的东西显示出来

    strock ：画线
    fill：填充
    CGContextStrokePath(ctr); //画线
    CGContextFillPath(ctr); //填充
    CGContextSetFillColorWithColor(ctr, [[UIColor blueColor] CGColor]);// 设置填充颜色
    CGContextSetStrokeColorWithColor(ctr, [[UIColor redColor] CGColor]);// 设置线的颜色
    同时设置线的样式和内容样式只能用CGContextDrawPath(ctr, CGPathDrawingMode);
    typedef CF_ENUM (int32_t, CGPathDrawingMode) {

    kCGPathFill, // 填充
kCGPathEOFill,
kCGPathStroke,// 画线
kCGPathFillStroke,// 画线和填充
kCGPathEOFillStroke

    };

    画一个正方形 用CGContextXX画一个正方形

  • (void)drawRect:(CGRect)rect {
    // 1.获取图形上下文
    CGContextRef ctr = UIGraphicsGetCurrentContext();
    CGContextMoveToPoint(ctr, 50, 50);
    CGContextAddLineToPoint(ctr, 100, 50);
    CGContextAddLineToPoint(ctr, 100, 100);
    CGContextAddLineToPoint(ctr, 50, 100);
    CGContextAddLineToPoint(ctr, 50, 50);

    // CGContextStrokePath(ctr);
    CGContextSetLineWidth(ctr, 10);
    // CGContextStrokePath(ctr);
    CGContextSetFillColorWithColor(ctr, [[UIColor blueColor] CGColor]);
    // [[UIColor redColor] setFill];
    CGContextSetStrokeColorWithColor(ctr, [[UIColor redColor] CGColor]);
    // CGContextFillPath(ctr);
    CGContextDrawPath(ctr, kCGPathEOFillStroke);//设置画图样式，划线，还是填充，还是既划线也填充
    }
